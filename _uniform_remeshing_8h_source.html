138  void partition_mesh_morton()
139  {

147

157  {
158  int order;
159  Resorting::MortonCode64 morton;
160  };
161 
162  std::vector<sortstruct> list_v;
163  list_v.resize(V_v.size());
164  const int multi = 1000;
165  // since the morton code requires a correct scale of input vertices,
166  // we need to scale the vertices if their coordinates are out of range
167  std::vector<Eigen::Vector3d> V = V_v; // this is for rescaling vertices
168  Eigen::Vector3d vmin, vmax;
169  vmin = V.front();
170  vmax = V.front();
171 
172  for (size_t j = 0; j < V.size(); j++) {
173  for (int i = 0; i < 3; i++) {
174  vmin(i) = std::min(vmin(i), V[j](i));
175  vmax(i) = std::max(vmax(i), V[j](i));
176  }
177  }
178 
179  Eigen::Vector3d center = (vmin + vmax) / 2;
180 
181  tbb::parallel_for(tbb::blocked_range<int>(0, V.size()), [&](tbb::blocked_range<int> r) {
182  for (int i = r.begin(); i < r.end(); i++) {
183  V[i] = V[i] - center;
184  }
185  });
186 
187  Eigen::Vector3d scale_point =
188  vmax - center; // after placing box at origin, vmax and vmin are symetric.
189 
190  double xscale, yscale, zscale;
191  xscale = fabs(scale_point[0]);
192  yscale = fabs(scale_point[1]);
193  zscale = fabs(scale_point[2]);
194  double scale = std::max(std::max(xscale, yscale), zscale);
195  if (scale > 300) {
196  tbb::parallel_for(
197  tbb::blocked_range<int>(0, V.size()),
198  [&](tbb::blocked_range<int> r) {
199  for (int i = r.begin(); i < r.end(); i++) {
200  V[i] = V[i] / scale;
201  }
202  });
203  }
204 
205  tbb::parallel_for(tbb::blocked_range<int>(0, V.size()), [&](tbb::blocked_range<int> r) {
206  for (int i = r.begin(); i < r.end(); i++) {
207  list_v[i].morton = Resorting::MortonCode64(
208  int(V[i][0] * multi),
209  int(V[i][1] * multi),
210  int(V[i][2] * multi));
211  list_v[i].order = i;
212  }
213  });

216  return (a.morton < b.morton);
217  };
218 
219  tbb::parallel_sort(list_v.begin(), list_v.end(), morton_compare);
220
