52  template <class T, int order>
53  inline T quadrature_error_1pixel_eval(
54  const Eigen::Matrix<T, 2, 3> edge_verts,
55  wmtk::LineQuadrature& quad) const
56  {
57  quad.get_quadrature(order);
58  T ret = static_cast<T>(0.0);
59  auto v1z = edge_verts(0, 2);
60  auto v2z = edge_verts(1, 2);
61  Eigen::Matrix<T, 1, 2> v12d, v22d;
62  v12d << edge_verts(0, 0), edge_verts(0, 1);
63  v22d << edge_verts(1, 0), edge_verts(1, 1);
64  // now do 1d quadrature
65  for (int i = 0; i < quad.points.rows(); i++) {
66  auto tmpu =
67  (1 - quad.points(i, 0)) * edge_verts(0, 0) + quad.points(i, 0) * edge_verts(1, 0);
68  auto tmpv =
69  (1 - quad.points(i, 0)) * edge_verts(0, 1) + quad.points(i, 0) * edge_verts(1, 1);
70  auto tmph = get(tmpu, tmpv);
71  auto tmpz = static_cast<T>(1 - quad.points(i, 0)) * v1z +
72  static_cast<T>(quad.points(i, 0)) * v2z;
73  ret += static_cast<T>(quad.weights(i)) * abs(tmph - tmpz);
74  }
75  auto edge_norm = sqrt(
76  (v12d(0) - v22d(0)) * (v12d(0) - v22d(0)) + (v12d(1) - v22d(1)) * (v12d(1) - v22d(1)));
77  return ret * edge_norm;
78  }

94  template <class T>
95  inline T get_error_per_edge_T(
96  const Eigen::Matrix<T, 2, 1>& p1,
97  const Eigen::Matrix<T, 2, 1>& p2) const
98  {
99  auto v1z = get(p1(0), p1(1));
100  auto v2z = get(p2(0), p2(1));
101  // get the pixel index of p1 and p2
102  auto get_coordinate = [&](const T& x, const T& y) -> std::pair<int, int> {
103  auto [xx, yy] = m_image.get_pixel_index(get_value(x), get_value(y));
104  return {m_image.get_coordinate(xx, m_image.get_wrapping_mode_x()),
105  m_image.get_coordinate(yy, m_image.get_wrapping_mode_y())};
106  };
107  auto [xx1, yy1] = get_coordinate(p1(0), p1(1));
108  auto [xx2, yy2] = get_coordinate(p2(0), p2(1));
109  // get all the pixels in between p1 and p2
110  auto pixel_num = std::max(abs(xx2 - xx1), abs(yy2 - yy1));
111  if (pixel_num <= 0) return static_cast<T>(0.);
112  assert(pixel_num > 0);
113  T error = static_cast<T>(0.0);
114  for (int i = 0; i < pixel_num; i++) {
115  const auto r0 = static_cast<T>(i) / pixel_num;
116  auto tmp_p1 = p1 * (1. - r0) + p2 * r0;
117  auto tmp_v1z = v1z * (1. - r0) + v2z * r0;
118  const auto r1 = static_cast<T>(i + 1) / pixel_num;
119  auto tmp_p2 = p1 * (1. - r1) + p2 * r1;
120  auto tmp_v2z = v1z * (1. - r1) + v2z * r1;
121  Eigen::Matrix<T, 2, 3> edge_verts;
122  edge_verts.row(0) << tmp_p1(0), tmp_p1(1), tmp_v1z;
123  edge_verts.row(1) << tmp_p2(0), tmp_p2(1), tmp_v2z;
124  wmtk::LineQuadrature quad;
125  auto displaced_error_per_pixel = quadrature_error_1pixel_eval<T, 5>(edge_verts, quad);
126  error += displaced_error_per_pixel;
127  }
128  assert(error >= 0);
129  return error;
130  }
