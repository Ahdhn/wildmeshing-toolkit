257  int find_local_edge(size_t v1_id, size_t v2_id) const
258  {
259  std::array<int, 2> e;
260  for (int j = 0; j < 4; j++) {
261  if (v1_id == m_indices[j])
262  e[0] = j;
263  else if (v2_id == m_indices[j])
264  e[1] = j;
265  }
266  if (e[0] > e[1]) std::swap(e[0], e[1]);
267  int i =
268  std::find(m_local_edges.begin(), m_local_edges.end(), e) - m_local_edges.begin();
269  if (i >= m_local_edges.size()) return -1;
270  return i;
271  }

273  int find_local_face(size_t v1_id, size_t v2_id, size_t v3_id) const
274  {
275  std::array<int, 3> f;
276  for (int j = 0; j < 4; j++) {
277  if (v1_id == m_indices[j])
278  f[0] = j;
279  else if (v2_id == m_indices[j])
280  f[1] = j;
281  else if (v3_id == m_indices[j])
282  f[2] = j;
283  }
284  std::sort(f.begin(), f.end());
285  int i =
286  std::find(m_local_faces.begin(), m_local_faces.end(), f) - m_local_faces.begin();
287  if (i >= m_local_edges.size()) return -1;
