259  int find_local_edge(size_t v1_id, size_t v2_id) const
260  {
261  std::array<int, 2> e;
262  for (int j = 0; j < 4; j++) {
263  if (v1_id == m_indices[j])
264  e[0] = j;
265  else if (v2_id == m_indices[j])
266  e[1] = j;
267  }
268  if (e[0] > e[1]) std::swap(e[0], e[1]);
269  int i =
270  std::find(m_local_edges.begin(), m_local_edges.end(), e) - m_local_edges.begin();
271  if (i >= m_local_edges.size()) return -1;
272  return i;
273  }
274 
275  int find_local_face(size_t v1_id, size_t v2_id, size_t v3_id) const
276  {
277  std::array<int, 3> f;
278  for (int j = 0; j < 4; j++) {
279  if (v1_id == m_indices[j])
280  f[0] = j;
281  else if (v2_id == m_indices[j])
282  f[1] = j;
283  else if (v3_id == m_indices[j])
284  f[2] = j;
285  }
286  std::sort(f.begin(), f.end());
287  int i =
288  std::find(m_local_faces.begin(), m_local_faces.end(), f) - m_local_faces.begin();
289  if (i >= m_local_edges.size()) return -1;
